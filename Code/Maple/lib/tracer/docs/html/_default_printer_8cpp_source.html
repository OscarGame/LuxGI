53 void DefaultPrinter::calcMaxNameLengths() {
54  auto *frames = trace->getFrames();
55  for (auto &i : *frames) {
56  string module = i.moduleName;
57  string fileName = i.fileName;
58 
59 #if !DISABLE_STD_FILESYSTEM
60  try {
61  if (cfg.canonicalizePaths) {
62  module = fs::canonical(fs::path(module)).string();
63  fileName = fs::canonical(fs::path(fileName)).string();
64  }
65  } catch (...) {}
66 
67  if (cfg.shortenFiles)
68  fileName = fs::path(fileName).filename().string();
69 
70  if (cfg.shortenModules)
71  module = fs::path(module).filename().string();
72 #else
73  regex shortenRegex(".*/");
74 
75  if (cfg.shortenFiles)
76  fileName = regex_replace(fileName, shortenRegex, "");
77 
78  if (cfg.shortenModules)
79  module = regex_replace(module, shortenRegex, "");
80 #endif
81 
82  if (module.length() > maxModuleNameLegth)
83  maxModuleNameLegth = module.length();
84 
85  if (i.funcName.length() > maxFuncNameLegth)
86  maxFuncNameLegth = i.funcName.length();
87 
88  stringstream address;
89  address << hex << i.frameAddr;
90  if (address.str().length() > maxAddressLength)
91  maxAddressLength = address.str().length();
92 
93  string lineInfo = fileName + ":" + to_string(i.line) + ":" + to_string(i.column);
94  if (lineInfo.length() > maxLineInfoLength)
95  maxLineInfoLength = lineInfo.length();
96  }
97 
98  calculatedMaxLengths = true;
99 }

102 std::string DefaultPrinter::genStringForFrameIMPL(size_t frameNum) {
103  stringstream outStream;
104  auto * frames = trace->getFrames();
105 
106  if (frameNum >= frames->size())
107  return "<INVALID PARAMETER>";
108 
109  Frame &i = frames->at(frameNum);
110 
111  if (!calculatedMaxLengths)
112  calcMaxNameLengths();
113 
114  string address = "<N/A>";
115  string funcName = "<N/A>";
116  string lineInfo = "<N/A>";
117  string module = "<N/A>";
118 
119  string colorAddress = cfg.colorNotFound;
120  string colorFuncName = cfg.colorNotFound;
121  string colorLineInfo = cfg.colorNotFound;
122  string colorModule = cfg.colorNotFound;
123 
124 
125  string moduleP = i.moduleName;
126  string fileName = i.fileName;
127 
128 #if !DISABLE_STD_FILESYSTEM
129  try {
130  if (cfg.canonicalizePaths) {
131  moduleP = fs::canonical(fs::path(moduleP)).string();
132  fileName = fs::canonical(fs::path(fileName)).string();
133  }
134  } catch (...) {}
135 
136  if (cfg.shortenFiles)
137  fileName = fs::path(fileName).filename().string();
138 
139  if (cfg.shortenModules)
140  moduleP = fs::path(moduleP).filename().string();
141 #else
142  regex shortenRegex(".*/");
143 
144  if (cfg.shortenFiles)
145  fileName = regex_replace(fileName, shortenRegex, "");
146 
147  if (cfg.shortenModules)
148  moduleP = regex_replace(moduleP, shortenRegex, "");
149 #endif
150 
151  if ((i.flags & FrameFlags::HAS_ADDRESS) == FrameFlags::HAS_ADDRESS) {
152  stringstream addressStream;
153  addressStream << hex << "0x" << setfill('0') << setw(static_cast<int>(maxAddressLength)) << i.frameAddr;
154  address = addressStream.str();
155  colorAddress = cfg.colorAddress;
156  }
157 
158  if ((i.flags & FrameFlags::HAS_FUNC_NAME) == FrameFlags::HAS_FUNC_NAME) {
159  funcName = i.funcName;
160  colorFuncName = cfg.colorFuncName;
161  }
162 
163  if ((i.flags & FrameFlags::HAS_LINE_INFO) == FrameFlags::HAS_LINE_INFO) {
164  lineInfo = fileName + ":" + to_string(i.line) + ":" + to_string(i.column);
165  colorLineInfo = cfg.colorLineInfo;
166  }
167 
168  if ((i.flags & FrameFlags::HAS_MODULE_INFO) == FrameFlags::HAS_MODULE_INFO) {
169  module = moduleP;
170  colorModule = cfg.colorModule;
171  }
172 
173  outStream << setfill(' ') << left // Setup
174  << cfg.colorFrameNum << "#" << setw(4) << frameNum // Frame number
175  << cfg.prefix << colorFuncName << setw(static_cast<int>(maxFuncNameLegth)) << funcName // Function Name
176  << cfg.seper1 << colorLineInfo << setw(static_cast<int>(maxLineInfoLength)) << lineInfo // Line Info
177  << cfg.seper2 << colorModule << setw(static_cast<int>(maxModuleNameLegth)) << module // Module
178  << cfg.seper3 << colorAddress << address // Address
179  << cfg.suffix << endl;
180 
181  return outStream.str();
182 }
