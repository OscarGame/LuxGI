53 void DebugInfoBFD::findInSection(bfd *abfd, bfd_section *section, void *ctx) {
54  DebugInfoBFD *obj = reinterpret_cast<DebugInfoBFD *>(ctx);
55 
56  if (obj->ctx.found)
57  return; // Nothing left to do
58 
59  if ((bfd_get_section_flags(abfd, section) & SEC_ALLOC) == 0)
60  return;
61 
62  bfd_vma vma = bfd_get_section_vma(abfd, section);
63  bfd_size_type size = bfd_get_section_size(section);
64 
65  auto addressToUse = obj->ctx.addr;
66 
67  // Check boundaries
68  if (addressToUse < vma || addressToUse >= vma + size) {
69  addressToUse = obj->ctx.addr - obj->ctx.baseAddr; // Try again with adjusted values
70  if (addressToUse < vma || addressToUse >= vma + size) {
71  return;
72  }
73  }
74 
75  const char * fileName_CSTR = nullptr;
76  const char * funcName_CSTR = nullptr;
77  unsigned int lineNumber = 0;
78  unsigned int column = 0;
79 
80 
81  if (bfd_find_nearest_line_discriminator(abfd,
82  section,
83  obj->ctx.secSym->symbols,
84  addressToUse - vma,
85  &fileName_CSTR,
86  &funcName_CSTR,
87  &lineNumber,
88  &column) == 0) {
89  if (bfd_find_nearest_line_discriminator(abfd,
90  section,
91  obj->ctx.dynSecSym->symbols,
92  addressToUse - vma,
93  &fileName_CSTR,
94  &funcName_CSTR,
95  &lineNumber,
96  &column) != 0) {
97  return;
98  }
99  }
100 
101  obj->ctx.found = true;
102 
103  obj->ctx.line = static_cast<int>(lineNumber);
104  obj->ctx.col = static_cast<int>(column);
105 
106  if (fileName_CSTR)
107  obj->ctx.fileName = fileName_CSTR;
108 
109  if (funcName_CSTR) {
110  char demangled[constants::MAX_FUNC_NAME];
111  int status = 0;
112  size_t legth = sizeof(demangled);
113  abi::__cxa_demangle(funcName_CSTR, demangled, &legth, &status);
114 
115  if (status == 0) {
116  obj->ctx.funcName = demangled;
117  } else {
118  obj->ctx.funcName = funcName_CSTR;
119  }
120  }
121 }

129 bool DebugInfoBFD::processFrames(std::vector<Frame> &frames) {
130  if (!isBfdInit) {
131  bfd_init();
132  isBfdInit = true;
133  }
134 
135  for (auto &i : frames) {
136  Dl_info dlInfo;
137 
138  if (dladdr(reinterpret_cast<void *>(i.frameAddr), &dlInfo) == 0) {
139  continue;
140  }
141 
142  if (dlInfo.dli_sname) {
143  char demangled[constants::MAX_FUNC_NAME];
144  int status = 0;
145  size_t legth = sizeof(demangled);
146  abi::__cxa_demangle(dlInfo.dli_sname, demangled, &legth, &status);
147 
148  i.flags |= FrameFlags::HAS_FUNC_NAME;
149 
150  if (status == 0) {
151  i.funcName = demangled;
152  } else {
153  i.funcName = dlInfo.dli_sname;
154  }
155  }
156 
157  if (dlInfo.dli_fname) {
158  i.flags |= FrameFlags::HAS_MODULE_INFO;
159  i.moduleName = dlInfo.dli_fname;
160  } else {
161  continue; // dlInfo.dli_fname is required for the next step
162  }
163 
164  // Start getting line info
165  abfdRAII abfd;
166  asymbolRAII symbols;
167  asymbolRAII dynSymbols;
168 
169  abfd.abfd = bfd_openr(dlInfo.dli_fname, nullptr);
170  if (abfd() == nullptr)
171  continue;
172 
173  abfd()->flags |= BFD_DECOMPRESS;
174 
175  if (bfd_check_format(abfd(), bfd_object) != TRUE)
176  continue;
177 
178  if ((bfd_get_file_flags(abfd()) & HAS_SYMS) == 0)
179  continue; // No debug symbols?
180 
181  auto storageSize = bfd_get_symtab_upper_bound(abfd());
182  auto dynamicStorageSize = bfd_get_dynamic_symtab_upper_bound(abfd());
183 
184  if (storageSize <= 0 && dynamicStorageSize <= 0)
185  continue;
186 
187  if (storageSize > 0) {
188  symbols.symbols = static_cast<bfd_symbol **>(malloc(static_cast<size_t>(storageSize)));
189  bfd_canonicalize_symtab(abfd(), symbols());
190  }
191 
192  if (dynamicStorageSize > 0) {
193  dynSymbols.symbols = static_cast<bfd_symbol **>(malloc(static_cast<size_t>(dynamicStorageSize)));
194  bfd_canonicalize_dynamic_symtab(abfd(), dynSymbols());
195  }
196 
197  ctx = FindInSectionContext();
198 
199  ctx.addr = i.frameAddr - 4;
200  ctx.baseAddr = reinterpret_cast<Address>(dlInfo.dli_fbase);
201  ctx.secSym = &symbols;
202  ctx.dynSecSym = &dynSymbols;
203 
204  bfd_map_over_sections(abfd(), &findInSection, reinterpret_cast<void *>(this));
205 
206  if (ctx.found) {
207  if (!ctx.fileName.empty()) {
208  i.flags |= FrameFlags::HAS_LINE_INFO;
209  i.fileName = ctx.fileName;
210  i.line = ctx.line;
211  i.column = ctx.col;
212  }
213 
214  if (!ctx.funcName.empty()) {
215  i.flags |= FrameFlags::HAS_FUNC_NAME;
216  i.funcName = ctx.funcName;
217  }
218  }
219  }
220 
221  return true;
222 }
