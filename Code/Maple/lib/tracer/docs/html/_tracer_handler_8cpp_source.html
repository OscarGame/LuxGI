63 void TracerHandler::sigHandler(int sigNum) {
64  TracerHandler *th = TracerHandler::getTracer();
65  auto cfg = th->getConfig();
66 
67  if (cfg.callDefultHandlerWhenDone) {
68 #if defined(__GNUC__) || defined(__clang__)
69 #pragma clang diagnostic push
70 #pragma clang diagnostic ignored "-Wold-style-cast"
71 #endif
72  signal(sigNum, SIG_DFL);
73 #if defined(__GNUC__) || defined(__clang__)
74 #pragma clang diagnostic pop
75 #endif
76  }
77 
78  auto tracerEngines = Tracer::getAvailableEngines();
79  auto debuggerEngines = Tracer::getAvailableDebuggers();
80 
81  TraceerEngines tEngine = tracerEngines[0];
82  DebuggerEngines tDebuger = debuggerEngines[0];
83 
84  for (auto i : cfg.preferredTracerEngines) {
85  for (auto j : tracerEngines) {
86  if (i == j) {
87  tEngine = i;
88  break;
89  }
90  }
91  }
92 
93  for (auto i : cfg.preferredDebuggerEngines) {
94  for (auto j : debuggerEngines) {
95  if (i == j) {
96  tDebuger = i;
97  break;
98  }
99  }
100  }
101 
102  Tracer t(tEngine, tDebuger);
103 
104  t.trace();
105 
106  AbstractPrinter * p = th->printer.get();
107  SystemInfoPrinter *sysP = nullptr;
108  if (!p) {
109  cerr << "[TRACER] (sigHandler) Fatal error: Printer not set (this code should never be executed)" << endl;
110  goto error;
111  }
112 
113  p->setTrace(&t);
114  sysP = dynamic_cast<SystemInfoPrinter *>(p);
115  if (sysP) {
116  sysP->setSignum(sigNum);
117  }
118 
119  if (cfg.autoPrintToStdErr) {
120  p->printToStdErr();
121  }
122 
123  if (cfg.autoPrintToFile) {
124  p->printToFile(cfg.logFile, cfg.appendToFile);
125  }
126 
127  if (cfg.callback)
128  cfg.callback(&t, p, cfg.callbackData);
129 
130 error:
131  if (!cfg.callDefultHandlerWhenDone)
132  return;
133 
134 #ifndef _WIN32
135  kill(getpid(), sigNum);
136 #endif
137 }
